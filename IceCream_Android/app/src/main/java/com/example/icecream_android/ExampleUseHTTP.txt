private void fetchData() {
        //Адреса
        String url = "https://b6ca-212-111-203-154.ngrok-free.app/database";
        //Створюємо Json
        JSONObject requestJson = new JSONObject();
        try {
            requestJson.put("function_name", "get_flavor_for_company");
            requestJson.put("param_dict", new JSONObject()
                    .put("point", "f7c654aa-60de-40c1-ac03-466a31882698")
                    .put("cone_name", new JSONArray().put("Солодкий ріжок").put("Картонний"))
                    .put("number", 10)
            );
        } catch (Exception e) {
            Toast.makeText(getContext(), "Помилка формування JSON: " + e.getMessage(), Toast.LENGTH_LONG).show();
            return;
        }
        
        //Створюємо об'єкт класу і надсилаємо
        HttpClientHelper clientHelper = new HttpClientHelper();
        clientHelper.sendPostRequest(url, requestJson, new HttpClientHelper.ResponseCallback() {
            //Обробка позитивного результату
            @Override
            public void onSuccess(String response) {
                try {
                    //Результат повертається json. Обробляємо його, як потрібно
                    JSONObject jsonResponse = new JSONObject(response);
                    JSONArray results = jsonResponse.getJSONArray("result");

                    for (int i = 0; i < results.length(); i++) {
                        JSONArray item = results.getJSONArray(i);
                        String name = item.getString(0);
                        double price = item.getDouble(1);
                        int quantity = item.getInt(2);

                        // Прямий вивід у консоль
                        System.out.println("Назва: " + name + ", Ціна: " + price + ", Кількість: " + quantity);
                    }

                    if (getActivity() != null) {
                        getActivity().runOnUiThread(() ->
                                Toast.makeText(getContext(), "Дані успішно отримано! Перевірте термінал.", Toast.LENGTH_LONG).show()
                        );
                    }
                } catch (Exception e) {
                    if (getActivity() != null) {
                        getActivity().runOnUiThread(() ->
                                Toast.makeText(getContext(), "Помилка парсингу JSON", Toast.LENGTH_LONG).show()
                        );
                    }
                }
            }


            //Обробка негативного результату
            @Override
            public void onFailure(String error) {
                if (getActivity() != null) {
                    getActivity().runOnUiThread(() ->
                            Toast.makeText(getContext(), "Помилка запиту: " + error, Toast.LENGTH_LONG).show()
                    );
                }
            }
        });
    }
